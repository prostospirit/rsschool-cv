      # Fastest way to sort array in Python
      def qsort(inlist): 
      if inlist == []: 
          return []
      else:
          pivot = inlist[0]
          lesser = qsort([x for x in inlist[1:] if x < pivot])
          greater = qsort([x for x in inlist[1:] if x >= pivot])
          return lesser + [pivot] + greater
